Reference code:
```ts
/**
 * # DROPDOWN COMPONENT
 * - Adds Bootstrap-like dropdown behavior to `.c-dropdown` markup.
 * - Multiple dropdowns can exist in the same root.
 * - Clicking outside, pressing Escape, or choosing an item closes the menu.
 */

const DROPDOWN_CLASS = 'c-dropdown';
const DROPDOWN_OPEN_CLASS = 'c-dropdown-open';
const DROPDOWN_TRIGGER_CLASS = 'c-dropdown-trigger';
const DROPDOWN_MENU_CLASS = 'c-dropdown-menu';
const DROPDOWN_ITEM_CLASS = 'c-dropdown-item';
const DROPDOWN_CLOSE_ANIMATION_MS = 160;

const initializedDropdownElements = new Set<HTMLElement>();
const dropdownComponentByElement = new WeakMap<HTMLElement, DropdownComponent>();
const closeTimerByElement = new WeakMap<HTMLElement, number>();
let areDocumentEventsBound = false;


/**
 * ## Dropdown Component
 * - Public controller for one dropdown instance.
 */
export interface DropdownComponent {
	// Closes the dropdown menu.
	close: () => void;

	// Removes listeners owned by this dropdown.
	destroy: () => void;
}

/**
 * Finds every dropdown in a root and wires reusable dropdown behavior.
 */
export function initializeDropdownComponents(root: ParentNode = document): DropdownComponent[] {
	const dropdownElements = getDropdownElements(root);
	const dropdowns = dropdownElements.map((dropdownElement) => initializeDropdownComponent(dropdownElement));

	if (dropdowns.length > 0) {
		bindDocumentEvents();
	}

	return dropdowns;
}

/**
 * Finds dropdowns, including the root itself when a single dropdown is passed in.
 */
function getDropdownElements(root: ParentNode): HTMLElement[] {
	const childDropdownElements = Array.from(root.querySelectorAll<HTMLElement>(`.${DROPDOWN_CLASS}`));

	if (root instanceof HTMLElement && root.classList.contains(DROPDOWN_CLASS)) {
		return [root, ...childDropdownElements];
	}

	return childDropdownElements;
}

/**
 * Wires one dropdown wrapper.
 */
function initializeDropdownComponent(dropdownElement: HTMLElement): DropdownComponent {
	const existingDropdown = dropdownComponentByElement.get(dropdownElement);

	if (existingDropdown) {
		return existingDropdown;
	}

	const trigger = dropdownElement.querySelector<HTMLElement>(`.${DROPDOWN_TRIGGER_CLASS}`);
	const menu = dropdownElement.querySelector<HTMLElement>(`.${DROPDOWN_MENU_CLASS}`);

	if (!trigger || !menu) {
		throw new Error('Required dropdown DOM element is missing.');
	}

	trigger.setAttribute('aria-expanded', String(isDropdownOpen(dropdownElement)));
	menu.hidden = !isDropdownOpen(dropdownElement);

	function onTriggerClick(event: MouseEvent): void {
		if (trigger instanceof HTMLAnchorElement) {
			event.preventDefault();
		}

		event.stopPropagation();
		toggleDropdown(dropdownElement);
	}

	function onMenuClick(event: MouseEvent): void {
		const target = event.target;

		if (!(target instanceof Element)) {
			return;
		}

		if (!target.closest(`.${DROPDOWN_ITEM_CLASS}`)) {
			return;
		}

		closeDropdown(dropdownElement);
	}

	trigger.addEventListener('click', onTriggerClick);
	menu.addEventListener('click', onMenuClick);
	initializedDropdownElements.add(dropdownElement);

	const dropdown: DropdownComponent = {
		close(): void {
			closeDropdown(dropdownElement);
		},
		destroy(): void {
			trigger.removeEventListener('click', onTriggerClick);
			menu.removeEventListener('click', onMenuClick);
			closeDropdown(dropdownElement);
			clearDropdownCloseTimer(dropdownElement);
			initializedDropdownElements.delete(dropdownElement);
			dropdownComponentByElement.delete(dropdownElement);

			if (!hasInitializedDropdowns()) {
				unbindDocumentEvents();
			}
		},
	};

	dropdownComponentByElement.set(dropdownElement, dropdown);

	return dropdown;
}

/**
 * Toggles the selected dropdown and closes sibling dropdowns.
 */
function toggleDropdown(dropdownElement: HTMLElement): void {
	if (isDropdownOpen(dropdownElement)) {
		closeDropdown(dropdownElement);
		return;
	}

	closeAllDropdowns(dropdownElement);
	openDropdown(dropdownElement);
}

/**
 * Opens one dropdown and syncs trigger accessibility state.
 */
function openDropdown(dropdownElement: HTMLElement): void {
	const menu = getDropdownMenu(dropdownElement);

	if (!menu) {
		return;
	}

	clearDropdownCloseTimer(dropdownElement);
	menu.hidden = false;
	dropdownElement.classList.add(DROPDOWN_OPEN_CLASS);
	setDropdownExpanded(dropdownElement, true);
}

/**
 * Closes one dropdown and syncs trigger accessibility state.
 */
function closeDropdown(dropdownElement: HTMLElement): void {
	const menu = getDropdownMenu(dropdownElement);

	dropdownElement.classList.remove(DROPDOWN_OPEN_CLASS);
	setDropdownExpanded(dropdownElement, false);

	if (!menu) {
		return;
	}

	clearDropdownCloseTimer(dropdownElement);

	const closeTimer = window.setTimeout(() => {
		menu.hidden = true;
		closeTimerByElement.delete(dropdownElement);
	}, DROPDOWN_CLOSE_ANIMATION_MS);

	closeTimerByElement.set(dropdownElement, closeTimer);
}

/**
 * Closes all initialized dropdowns except the optional ignored one.
 */
function closeAllDropdowns(ignoredDropdownElement?: HTMLElement): void {
	getInitializedDropdowns().forEach((dropdownElement) => {
		if (dropdownElement === ignoredDropdownElement) {
			return;
		}

		closeDropdown(dropdownElement);
	});
}

/**
 * Closes dropdowns when the next document click lands outside them.
 */
function closeDropdownsOnOutsideClick(event: MouseEvent): void {
	const target = event.target;

	if (!(target instanceof Node)) {
		return;
	}

	getInitializedDropdowns().forEach((dropdownElement) => {
		if (dropdownElement.contains(target)) {
			return;
		}

		closeDropdown(dropdownElement);
	});
}

/**
 * Closes dropdowns on Escape like Bootstrap dropdowns.
 */
function closeDropdownsOnEscape(event: KeyboardEvent): void {
	if (event.key !== 'Escape') {
		return;
	}

	closeAllDropdowns();
}

/**
 * Updates the trigger's expanded state.
 */
function setDropdownExpanded(dropdownElement: HTMLElement, isExpanded: boolean): void {
	const trigger = dropdownElement.querySelector<HTMLElement>(`.${DROPDOWN_TRIGGER_CLASS}`);

	if (!trigger) {
		return;
	}

	trigger.setAttribute('aria-expanded', String(isExpanded));
}

/**
 * Checks whether the dropdown menu is open.
 */
function isDropdownOpen(dropdownElement: HTMLElement): boolean {
	return dropdownElement.classList.contains(DROPDOWN_OPEN_CLASS);
}

/**
 * Finds the menu inside one dropdown.
 */
function getDropdownMenu(dropdownElement: HTMLElement): HTMLElement | null {
	return dropdownElement.querySelector<HTMLElement>(`.${DROPDOWN_MENU_CLASS}`);
}

/**
 * Clears a pending close timer before a dropdown opens or is destroyed.
 */
function clearDropdownCloseTimer(dropdownElement: HTMLElement): void {
	const closeTimer = closeTimerByElement.get(dropdownElement);

	if (!closeTimer) {
		return;
	}

	window.clearTimeout(closeTimer);
	closeTimerByElement.delete(dropdownElement);
}

/**
 * Reads initialized dropdowns from the document.
 */
function getInitializedDropdowns(): HTMLElement[] {
	return Array.from(initializedDropdownElements);
}

/**
 * Checks whether any dropdowns still exist in the document.
 */
function hasInitializedDropdowns(): boolean {
	return getInitializedDropdowns().length > 0;
}

/**
 * Binds shared document listeners once for all dropdowns.
 */
function bindDocumentEvents(): void {
	if (areDocumentEventsBound) {
		return;
	}

	document.addEventListener('click', closeDropdownsOnOutsideClick);
	document.addEventListener('keydown', closeDropdownsOnEscape);
	areDocumentEventsBound = true;
}

/**
 * Unbinds shared document listeners when the last dropdown is destroyed.
 */
function unbindDocumentEvents(): void {
	document.removeEventListener('click', closeDropdownsOnOutsideClick);
	document.removeEventListener('keydown', closeDropdownsOnEscape);
	areDocumentEventsBound = false;
}
```

	01-01. Add document `click` and `keydown` listeners only once.
